cmake_minimum_required(VERSION 2.8)
project(plugins)

    add_subdirectory(chisquare)

if(${TLP_BUILD_LEVENBERG_MARQUARDT_PLUGIN} STREQUAL ON)
    add_subdirectory(levenberg_marquardt)

    add_subdirectory(nelder_mead)
